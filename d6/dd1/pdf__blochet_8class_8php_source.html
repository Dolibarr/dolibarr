345  $yp = 0;
346  $lineinpage = 0;
347  $num = count($this->lines);

349  // Dynamic max line heigh calculation

351  $dynamic_line_height[] = $pdf->getStringHeight(60, $outputlangs->convToOutputCharset($this->lines[$j]->bank_chq));
