531 function quotedPrintEncode($str, $forcal = 0)
532 {
533  $lines = preg_split("/\r\n/", $str);
534  $out = "";
535 
536  foreach ($lines as $line) {
537  $newpara = "";
538 
539  // Do not use dol_strlen here, we need number of bytes
540  $strlength = strlen($line);
541 
542  for ($j = 0; $j < $strlength; $j++) {
543  $char = substr($line, $j, 1);
544  $ascii = ord($char);
545 
546  if ($ascii < 32 || $ascii === 61 || $ascii > 126) {
547  $char = "=".strtoupper(sprintf("%02X", $ascii));
548  }
549 
550  // Do not use dol_strlen here, we need number of bytes
551  if ((strlen($newpara) + strlen($char)) >= 76) {
552  // New line with carray-return (CR) and line-feed (LF)
553  $out .= $newpara."=\r\n";
554 
555  // extra space for cal
556  if ($forcal) {
557  $out .= " ";
558  }
559 
560  $newpara = "";
561  }
562 
563  $newpara .= $char;
564  }
565 
566  $out .= $newpara;
567  }
568  return trim($out);
569 }
